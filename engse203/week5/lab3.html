Step 3: ThemeContext.jsx

Context สำหรับจัดการ Light/Dark theme และบันทึกลง LocalStorage

// src/contexts/ThemeContext.jsx
import { createContext, useContext, useState, useEffect } from 'react';

const ThemeContext = createContext();

export function ThemeProvider({ children }) {
  const [theme, setTheme] = useState(() => {
    const saved = localStorage.getItem('portfolio-theme');
    return saved || 'light';
  });

  useEffect(() => {
    localStorage.setItem('portfolio-theme', theme);
    document.documentElement.setAttribute('data-theme', theme);
  }, [theme]);

  const toggleTheme = () => {
    setTheme(prev => prev === 'light' ? 'dark' : 'light');
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
}

🌟 Git Workflow Development Process

เราจะใช้ Feature Branch Workflow ซึ่งเป็นกระบวนการที่เป็นที่นิยมในการพัฒนาซอฟต์แวร์ โดยจะสร้าง Branch ใหม่สำหรับแต่ละ Feature ที่จะทำ

main
Production Ready Code
feature/projects-section
พัฒนาส่วน Projects
feature/contact-form
พัฒนาส่วน Contact
feature/enhancements
ปรับปรุงและเพิ่มเติม
main
Merge features back

Phase 1: Initial Setup & Repository Creation

# 1. Initialize Git repository
git init

# 2. Create .gitignore (ควรทำเป็นอันดับแรกๆ)
echo "node_modules/\ndist/\n.env*" > .gitignore

# 3. Initial commit (หลังจากใส่โค้ดเริ่มต้นทั้งหมดแล้ว)
git add .
git commit -m "feat: initial project setup with starter code"

# 4. Create GitHub repository และ push
git branch -M main
git remote add origin https://github.com/[username]/[repository-name].git
git push -u origin main

Phase 2: Feature Development (ส่วนที่นักศึกษาต้องทำ)

สำหรับแต่ละ Feature ให้สร้าง Branch ใหม่จาก `main`

# ตัวอย่างการสร้าง branch สำหรับ feature แรก
git checkout -b feature/projects-section

# ... หลังจากพัฒนา feature เสร็จแล้ว ...
git add .
git commit -m "feat: implement projects section"
git push origin feature/projects-section

ทำซ้ำขั้นตอนเดิมสำหรับ feature/contact-form และ feature/enhancements

Phase 3: Integration และ Final Polish

เมื่อพัฒนาทุก Feature เสร็จสิ้น ให้ merge กลับเข้ามาที่ `main` (แนะนำให้ทำผ่าน Pull Request บน GitHub)

# กลับไปที่ main branch
git checkout main

# ดึงข้อมูลล่าสุด (เผื่อมีการเปลี่ยนแปลง)
git pull origin main

# Merge feature branch (ตัวอย่าง)
git merge feature/projects-section

# ... ทำการ merge ให้ครบทุก feature ...

# Push final version to main
git push origin main

ตัวอย่างโค้ด Projects.jsx ที่ต้องพัฒนา:

// src/components/Projects/Projects.jsx
import { useState } from 'react';
import ProjectCard from './ProjectCard'; // ต้องสร้าง Component นี้
import { projects } from '../../data/portfolioData';

function Projects() {
  const [filter, setFilter] = useState('all');
  
  const technologies = ['all', ...new Set(projects.flatMap(p => p.technologies))];
  
  const filteredProjects = filter === 'all' 
    ? projects 
    : projects.filter(p => p.technologies.includes(filter));
  
  return (
    <section id="projects">
      <h2>My Projects</h2>
      <div className="project-filters">
        {technologies.map(tech => (
          <button key={tech} onClick={() => setFilter(tech)}>
            {tech}
          </button>
        ))}
      </div>
      <div className="projects-grid">
        {filteredProjects.map(project => (
          <ProjectCard key={project.id} project={project} />
        ))}
      </div>
    </section>
  );
}

🚀 Build และ Deployment Guide

หลังจากพัฒนาเสร็จสิ้น ขั้นตอนสุดท้ายคือการ Build โปรเจกต์และนำขึ้น Host เพื่อให้สามารถเข้าชมออนไลน์ได้ แนะนำให้ใช้ Netlify หรือ Vercel เพราะง่ายและมีการเชื่อมต่อกับ Git ที่ดีเยี่ยม

Deploy ด้วย Netlify (Git Integration)

  1. Push โค้ดทั้งหมดขึ้น GitHub repository
  2. ไปที่ Netlify และสมัครสมาชิก
  3. คลิก "New site from Git"
  4. เชื่อมต่อกับ GitHub และเลือก Repository ของโปรเจกต์
  5. ตั้งค่า Build settings:
    • Build command: npm run build
    • Publish directory: dist
  6. คลิก "Deploy site" และรอสักครู่ เว็บไซต์ของคุณก็จะออนไลน์!

Deploy ด้วย Vercel (Git Integration)

  1. Push โค้ดทั้งหมดขึ้น GitHub repository
  2. ไปที่ Vercel และสมัครสมาชิก
  3. คลิก "Import Project" และเลือก "Import Git Repository"
  4. เชื่อมต่อกับ GitHub และเลือก Repository ของโปรเจกต์
  5. Vercel จะตรวจจับว่าเป็น React Project (Vite) โดยอัตโนมัติ
  6. คลิก "Deploy" และรอสักครู่

Deploy ด้วย GitHub Pages

  1. ติดตั้ง gh-pages: 
    npm install --save-dev gh-pages
  2. แก้ไขไฟล์ vite.config.js เพิ่ม base: 
    // vite.config.js
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'

export default defineConfig({
  plugins: [react()],
  base: '/[repository-name]/', // <-- เพิ่มส่วนนี้
})
  3. เพิ่ม script ใน package.json: 
    "scripts": {
  // ... other scripts
  "predeploy": "npm run build",
  "deploy": "gh-pages -d dist"
}
  4. สั่ง Deploy: 
    npm run deploy
  5. ไปที่ GitHub repository > Settings > Pages > Source และเลือก branch เป็น gh-pages

🆘 Troubleshooting Guide

เมื่อเจอปัญหา ลองดูวิธีแก้ไขเบื้องต้นเหล่านี้ก่อน

  • Build Errors: ลองลบ node_modules และ package-lock.json แล้วติดตั้งใหม่ด้วย npm install
  • Theme ไม่ทำงาน: ตรวจสอบว่า ThemeProvider ครอบ App ใน main.jsx หรือไม่ และเช็ค CSS Variables
  • Git มีปัญหา: หากต้องการย้อนกลับไป commit ก่อนหน้า ใช้ git reset --hard HEAD~1 (ระวัง! การเปลี่ยนแปลงล่าสุดจะหายไป)
