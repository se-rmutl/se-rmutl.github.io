// src/utils.js
export function getStatusLabel(status) {
  const labels = {
    todo: "To do",
    doing: "In progress",
    done: "Done",
  };
  return labels[status] ?? "Unknown";
}
// src/main.js
import { getStatusLabel } from "./utils.js";

Error Handling Pattern

try / catch / finally ใช้อย่างไร?

try

เริ่มโหลดข้อมูลและ render เมื่อสำเร็จ

response.ok

fetch ไม่ throw เองเมื่อ HTTP 404/500 จึงต้องตรวจเอง

catch

รับ error และแสดงข้อความที่ผู้ใช้เข้าใจได้

finally

ทำงานเก็บกวาดเสมอ เช่น คืนสถานะปุ่ม

try {
  setMessage("loading", "กำลังโหลดข้อมูล...");
  const tasks = await fetchLearningTasks();
  renderDashboard(tasks);
  setMessage("success", `โหลดข้อมูล ${tasks.length} รายการแล้ว`);
} catch (error) {
  setMessage("error", `ไม่สามารถโหลดข้อมูลได้: ${error.message}`);
} finally {
  console.log("Load attempt finished");
}

async function fetchLearningTasks({ simulateError=false }={}) {
  if (simulateError) throw new Error("data source is unavailable");

  const response = await fetch(blobJsonUrl);
  if (!response.ok) throw new Error(`HTTP ${response.status}`);
  return response.json();
}

Why Vite?

Vite ช่วยแก้ความยุ่งยาก 3 เรื่องในงานเรา

🖥️

1. Local Dev Server

เปิดเว็บผ่าน http://localhost:5173 แทนการดับเบิลคลิก index.html จึงทดสอบ ES Modules และ fetch ในบริบท HTTP ที่ถูกต้อง

⚡

2. HMR

เมื่อแก้ไฟล์และ Save Vite จะอัปเดตส่วนที่เปลี่ยนให้เร็วมาก โดยพยายามไม่ reload ทั้งหน้า

ใน Vanilla JS อย่าคาดหวังว่า state จะคงอยู่ทุกกรณี เพราะ module อาจถูกเรียกใหม่ได้

🧩

3. Developer Experience

ช่วยจัดการ dependency และ asset เช่น import "./style.css" ใน main.js ได้สะดวก

ประเด็นสำคัญ: ปัญหาไม่ใช่ “JavaScript ทำไม่ได้” แต่คือการเปิดผ่าน file:// อาจถูกจำกัดโดยนโยบายความปลอดภัยของเบราว์เซอร์ และไม่เหมือนเว็บที่ deploy จริง

Trade-off

“ไม่ใช้ Vite ได้ไหม?” — ได้ แต่ต้องจัดการเองมากขึ้น

เรื่องใช้ Vite ใน ENGSE203Pure Vanilla JS ไม่ใช้ Vite
เปิดเทสในเครื่องnpm run dev เปิด local server ให้ทันทีต้องใช้ Live Server หรือจัด server เอง
ES Modules / fetchทดสอบภายใต้ HTTP และใกล้เคียงเว็บจริงต้องระวัง file:// และ path เอง
จัดการ CSS / assetimport จาก JS และให้ Vite ติดตาม dependencyเชื่อม CSS / asset ด้วย HTML และดูแล path เอง
เตรียม deploynpm run build สร้าง asset ที่ optimize แล้วอัปโหลดไฟล์ต้นฉบับและจัดการ optimization เอง
GitHub Pagesกำหนด base และ output folder ชัดเจนต้องตรวจ path ของ module / asset ทุกจุดเอง
เหตุผลที่เราใช้ Vite ใน Week 02: ฝึก tooling ที่ใช้จริงในอุตสาหกรรม และปูพื้นก่อนเข้าสู่ React.js ในสัปดาห์ถัดไป

Before Submit

Checklist ก่อนส่ง LAB 2

  • อยู่ใน branch feature ระหว่างพัฒนา และ merge เข้า main แล้ว
  • มี ES Modules อย่างน้อย 4 ไฟล์
  • JSON โหลดด้วย fetch + async/await
  • มี error state เมื่อเปิด ?simulateError=1
  • npm run check และ npm run build สำเร็จ
  • vite.config.js base path ตรงกับ repo
  • main branch มี docs/ ล่าสุด
  • GitHub Pages ตั้งค่า main /docs
  • README มีวิธีรัน, URL, screenshot และ AI disclosure
  • ไม่ commit node_modules, token, password, .env
Common Problems

ปัญหาที่พบบ่อยและวิธีแก้

JSON ไม่โหลด

อย่าเปิด index.html ด้วย file:// ให้ใช้ npm run dev และตรวจ path

GitHub Pages 404

ตรวจ base, docs/ บน main, Settings → Pages

import error

ตรวจ type="module" และ relative import เช่น ./utils.js

node_modules ติด git

เพิ่ม node_modules/ ใน .gitignore แล้วใช้ git rm -r --cached node_modules
