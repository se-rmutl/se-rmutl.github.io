import React, { useState, useEffect } from 'react';

function UserProfile() {
  // useState - จัดการ state ภายใน component

      <h1>{user?.name}</h1>
      <p>{user?.email}</p>
    </div>
  );
}

Simple Custom Hook Example JSX 
// Custom Hook สำหรับจัดการ counter
function useCounter(initialValue = 0) {
  const [count, setCount] = useState(initialValue);

  const increment = () => setCount(count + 1);
  const decrement = () => setCount(count - 1);
  const reset = () => setCount(initialValue);

  return { count, increment, decrement, reset };
}

Custom Hook ตัวอย่าง: useFetch

useFetch Hook - ดึงข้อมูลจาก API JSX 
import { useState, useEffect } from 'react';

function useFetch(url) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const fetchData = async () => {
      try {
        setLoading(true);
        const response = await fetch(url);
        
        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }
        
        const jsonData = await response.json();
        setData(jsonData);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, [url]); // Re-fetch เมื่อ URL เปลี่ยน

  return { data, loading, error };
}

// การใช้งาน useFetch
function UserList() {
  const { data: users, loading, error } = useFetch('/api/users');

  if (loading) return <div>กำลังโหลด...</div>;
  if (error) return <div>เกิดข้อผิดพลาด: {error}</div>;

  return (
    <ul>
      {users?.map(user => (
        <li key={user.id}>{user.name}</li>
      ))}
    </ul>
  );
}

Workshop Solution: useLocalStorage Hook

useLocalStorage Custom Hook JSX 
function useLocalStorage(key, initialValue) {
  // อ่านค่าเริ่มต้นจาก localStorage
  const [storedValue, setStoredValue] = useState(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(`Error loading localStorage key "${key}":`, error);
      return initialValue;
    }
  });

  // สร้าง setter function ที่บันทึกลง localStorage
  const setValue = (value) => {
    try {
      // Allow value to be a function (like useState)
      const valueToStore = value instanceof Function 
        ? value(storedValue) 
        : value;
      
      setStoredValue(valueToStore);
      window.localStorage.setItem(key, JSON.stringify(valueToStore));
    } catch (error) {
      console.error(`Error saving to localStorage key "${key}":`, error);
    }
  };

  return [storedValue, setValue];
}

// ตัวอย่างการใช้งาน
function ThemeSettings() {
  const [theme, setTheme] = useLocalStorage('theme', 'light');
  
  return (
    <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>
      Current theme: {theme}
    </button>
  );
}

Higher-Order Components (HOC)

HOC คืออะไร?

Higher-Order Component คือ function ที่รับ component เป็น parameter และ return component ใหม่ที่มี functionality เพิ่มเติม

HOC Pattern Example - withAuth JSX 
// HOC สำหรับตรวจสอบ authentication
function withAuth(WrappedComponent) {
  return function WithAuthComponent(props) {
    const [isAuthenticated, setIsAuthenticated] = useState(false);
    const [loading, setLoading] = useState(true);

    useEffect(() => {
      // ตรวจสอบ auth status
      const checkAuth = async () => {
        try {
          const token = localStorage.getItem('token');
          if (token) {
            // Verify token with backend
            const response = await fetch('/api/verify', {
              headers: { 'Authorization': `Bearer ${token}` }
            });
            setIsAuthenticated(response.ok);
          }
        } finally {
          setLoading(false);
        }
      };
      checkAuth();
    }, []);

    if (loading) return <div>Checking authentication...</div>;
    if (!isAuthenticated) return <div>Please login to continue</div>;

    // Pass all props to wrapped component
    return <WrappedComponent {...props} />;
  };
}

// การใช้งาน HOC
function Dashboard() {
  return <div>Welcome to Dashboard!</div>;
}

const ProtectedDashboard = withAuth(Dashboard);

// ใช้ใน App
<ProtectedDashboard />

Render Props Pattern

Render Props คืออะไร?

เทคนิคการแชร์ code ระหว่าง components โดยใช้ prop ที่เป็น function และ return JSX

Render Props Example - Mouse Tracker JSX 
// Component ที่ track mouse position
function MouseTracker({ render }) {
  const [position, setPosition] = useState({ x: 0, y: 0 });

  useEffect(() => {
    const handleMouseMove = (event) => {
      setPosition({ x: event.clientX, y: event.clientY });
    };

    window.addEventListener('mousemove', handleMouseMove);
    return () => window.removeEventListener('mousemove', handleMouseMove);
  }, []);

  // เรียก render prop function พร้อมส่ง data
  return render(position);
}

// การใช้งาน Render Props
function App() {
  return (
    <div>
      <h1>Mouse Position Tracker</h1>
      
      <MouseTracker 
        render={({ x, y }) => (
          <div>
            Mouse position: ({x}, {y})
          </div>
        )}
      />

      {/* ใช้ซ้ำกับ UI ที่ต่างกัน */}
      <MouseTracker 
        render={({ x, y }) => (
          <div style={{
            width: 20,
            height: 20,
            borderRadius: '50%',
            background: 'red',
            position: 'fixed',
            left: x - 10,
            top: y - 10
          }}/>
        )}
      />
    </div>
  );
}

Context API Implementation JSX 
// 1. สร้าง Context
const UserContext = React.createContext();

// 2. สร้าง Provider Component
function UserProvider({ children }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Fetch user data
    fetchUserData().then(data => {
      setUser(data);
      setLoading(false);
    });
  }, []);

  const login = async (email, password) => {
    const response = await fetch('/api/login', {
      method: 'POST',
      body: JSON.stringify({ email, password })
    });
    const userData = await response.json();
    setUser(userData);
  };

  const logout = () => setUser(null);

  const value = { user, loading, login, logout };

  return (
    <UserContext.Provider value={value}>
      {children}
    </UserContext.Provider>
  );
}

const counterSlice = createSlice({

Performance Optimization - React.memo

React.memo - Prevent Unnecessary Re-renders

Without React.memo (Problem) JSX 
// ❌ Child component re-renders ทุกครั้งที่ parent re-render
function ExpensiveList({ items }) {
  console.log('ExpensiveList rendered!'); // จะ log ทุกครั้ง!
  
  return (
    <ul>
      {items.map(item => (
        <li key={item.id}>{item.name}</li>
      ))}
    </ul>
  );
}

function Parent() {
  const [count, setCount] = useState(0);
  const items = [{ id: 1, name: 'Item 1' }]; // ไม่เปลี่ยน
  
  return (
    <div>
      <button onClick={() => setCount(count + 1)}>
        Count: {count}
      </button>
      <ExpensiveList items={items} /> {/* Re-render ทุกครั้ง! */}
    </div>
  );
}
With React.memo (Solution) JSX 
// ✅ ใช้ React.memo เพื่อ memoize component
const ExpensiveList = React.memo(function ExpensiveList({ items }) {
  console.log('ExpensiveList rendered!'); // จะ log เมื่อ items เปลี่ยนเท่านั้น
  
  return (
    <ul>
      {items.map(item => (
        <li key={item.id}>{item.name}</li>
      ))}
    </ul>
  );
});

// Custom comparison function (optional)
const MemoizedComponent = React.memo(Component, (prevProps, nextProps) => {
  // return true ถ้า props เท่ากัน (skip re-render)
  // return false ถ้า props ต่างกัน (re-render)
  return prevProps.id === nextProps.id;
});

Performance: useMemo & useCallback

useMemo - Memoize Values

useMemo Example JSX 
function ExpensiveComponent({ data, filter }) {
  // ❌ คำนวณใหม่ทุกครั้งที่ component re-render
  // const filteredData = data.filter(item => item.includes(filter));
  
  // ✅ คำนวณใหม่เฉพาะเมื่อ data หรือ filter เปลี่ยน
  const filteredData = useMemo(() => {
    console.log('Filtering data...');
    return data.filter(item => item.includes(filter));
  }, [data, filter]);

  const totalPrice = useMemo(() => {
    return filteredData.reduce((sum, item) => sum + item.price, 0);
  }, [filteredData]);

  return (
    <div>
      <p>Total: {totalPrice}</p>
      {filteredData.map(item => <Item key={item.id} data={item} />)}
    </div>
  );
}

useCallback - Memoize Functions

useCallback Example JSX 
function TodoList() {
  const [todos, setTodos] = useState([]);
  const [filter, setFilter] = useState('');

  // ❌ สร้าง function ใหม่ทุกครั้ง = child component re-render
  // const handleDelete = (id) => {
  //   setTodos(todos.filter(todo => todo.id !== id));

    setTodos(prevTodos => [...prevTodos, { id: Date.now(), text }]);
  }, []);

  return (
    <div>
      <AddTodo onAdd={handleAdd} />
      <TodoItems 
        todos={todos} 
        onDelete={handleDelete}
        filter={filter}
      />
    </div>
  );
}

API Service Configuration JSX 
// services/api.js
const API_URL = import.meta.env.VITE_API_URL || 'http://localhost:3000';

class ApiService {
  constructor() {
    this.baseURL = API_URL;
    this.token = localStorage.getItem('token');
  }

  async request(endpoint, options = {}) {
    const url = `${this.baseURL}${endpoint}`;
    
    const config = {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        ...options.headers,
      }
    };

    // Add auth token if exists
    if (this.token) {
      config.headers.Authorization = `Bearer ${this.token}`;
    }

    try {
      const response = await fetch(url, config);
      
      if (!response.ok) {
        if (response.status === 401) {
          // Handle unauthorized
          this.logout();
        }
        throw new Error(`HTTP error! status: ${response.status}`);
      }
      
      return await response.json();
    } catch (error) {
      console.error('API Error:', error);
      throw error;
    }
  }

  // Auth methods
  async login(email, password) {
    const data = await this.request('/auth/login', {
      method: 'POST',
      body: JSON.stringify({ email, password })
    });
    this.token = data.token;
    localStorage.setItem('token', data.token);
    return data;
  }

  logout() {
    this.token = null;
    localStorage.removeItem('token');
    window.location.href = '/login';
  }

  // CRUD operations
  get = (endpoint) => this.request(endpoint);
  
  post = (endpoint, data) => this.request(endpoint, {
    method: 'POST',
    body: JSON.stringify(data)
  });
  
  put = (endpoint, data) => this.request(endpoint, {
    method: 'PUT',
    body: JSON.stringify(data)
  });
  
  delete = (endpoint) => this.request(endpoint, {
    method: 'DELETE'
  });
}

export default new ApiService();
Using API Service in Components JSX 
import api from './services/api';

function useProducts() {
  const [products, setProducts] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  const fetchProducts = async () => {
    try {
      setLoading(true);
      const data = await api.get('/products');
      setProducts(data);
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };

  const createProduct = async (productData) => {
    const newProduct = await api.post('/products', productData);
    setProducts([...products, newProduct]);
    return newProduct;
  };

  const updateProduct = async (id, updates) => {
    const updated = await api.put(`/products/${id}`, updates);
    setProducts(products.map(p => p.id === id ? updated : p));
  };

  const deleteProduct = async (id) => {
    await api.delete(`/products/${id}`);
    setProducts(products.filter(p => p.id !== id));
  };

  useEffect(() => {
    fetchProducts();
  }, []);

  return {
    products,
    loading,
    error,
    createProduct,
    updateProduct,
    deleteProduct,
    refetch: fetchProducts
  };
}
