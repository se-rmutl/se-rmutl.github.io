Hour 2: Models & Controllers (70 นาที)

ลงมือเขียน Logic หลักของ API

ชั่วโมงนี้คือหัวใจของการลงมือเขียนโค้ด เราจะสร้าง Model เพื่อจำลองข้อมูล และ Controller เพื่อจัดการ Business Logic ทั้งหมด


// models/Agent.js
class Agent {
  constructor(data) {
    this.id = data.id || this.generateId();
    this.agentCode = data.agentCode;
    this.name = data.name;
    this.email = data.email;
    this.department = data.department || 'General';
    this.status = data.status || 'Available';
    this.lastStatusChange = new Date();
    this.statusHistory = [];
  }

  generateId() {
    return Date.now() + Math.random().toString(36).substr(2, 9);
  }

  updateStatus(newStatus, reason = null) {
    this.statusHistory.push({ from: this.status, to: newStatus, reason, timestamp: new Date() });
    this.status = newStatus;
    this.lastStatusChange = new Date();
  }

  toJSON() {
    return { id: this.id, agentCode: this.agentCode, name: this.name, email: this.email, department: this.department, status: this.status };
  }
}

Hour 3: Validation & Error Handling (70 นาที)

สร้าง API ที่ฉลาด ปลอดภัย และจัดการข้อผิดพลาดได้

ทำให้ API ของเราแข็งแกร่งและน่าเชื่อถือโดยการดักจับข้อมูลที่ไม่ถูกต้องและจัดการกับ Error ที่อาจเกิดขึ้นได้อย่างสวยงาม

เขียน Middleware สำหรับ Validation!

  • TODO #4: `statusUpdate Schema` - ใน `middleware/validation.js` ให้สร้าง Joi schema สำหรับ validate request body ของการอัปเดตสถานะ ต้องมี `status` ที่ถูกต้อง และ `reason` ที่เป็น string (optional).
  • TODO #5: `validateStatusUpdate` - สร้าง middleware function ที่ใช้ schema จาก TODO #4 เพื่อ validate request และส่ง error กลับหากข้อมูลไม่ถูกต้อง.

// middleware/errorHandler.js
const { sendError } = require('../utils/apiResponse');

const globalErrorHandler = (err, req, res, next) => {
  console.error('🚨 Global Error Handler:', err.message);
  return sendError(res, 'Internal server error', 500);
};

const notFoundHandler = (req, res) => {
  return sendError(res, `Route ${req.originalUrl} not found`, 404);
};

module.exports = { globalErrorHandler, notFoundHandler };
