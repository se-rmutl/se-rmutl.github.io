เป้าหมายหลัก (Goal)

สร้างเว็บแอปฯ ที่มีช่องโหว่ แล้วแก้ไขให้ปลอดภัย

เวลาที่คาดว่าจะใช้ (Time)

3 - 4 ชั่วโมง

เทคโนโลยี (Tech Stack)

Node.js, Express, MSSQL, HTML, CSS, JS

Roadmap: 4 ขั้นตอนสู่การเป็น Secure Developer

สร้าง & เจาะ (Build & Break)

ลงมือสร้างเว็บแอปฯ ที่มีช่องโหว่ตามมาตรฐาน OWASP Top 10 จากนั้นสวมบทบาทแฮกเกอร์เพื่อเจาะระบบของตัวเอง ทำความเข้าใจกลไกการโจมตี SQL Injection, XSS, IDOR และผลกระทบที่เกิดขึ้นจริง

แก้ไข & ป้องกัน (Fix & Fortify)

กลับมาในฐานะวิศวกรซอฟต์แวร์เพื่อแก้ไขช่องโหว่ทั้งหมด โดยใช้ Security Best Practices เช่น Prepared Statements, Input Sanitization, Authorization checks และการใช้ Security Headers เพื่อสร้างเกราะป้องกันที่แข็งแกร่ง

เปรียบเทียบ & วิเคราะห์ (Compare & Analyze)

ทดสอบและเปรียบเทียบผลลัพธ์ระหว่างเวอร์ชันที่มีช่องโหว่ (Vulnerable) และเวอร์ชันที่ปลอดภัย (Secure) เพื่อให้เห็นความแตกต่างอย่างชัดเจน และเข้าใจถึงความสำคัญของการเขียนโค้ดที่ปลอดภัย

รายงาน & สรุปผล (Report & Reflect)

จัดทำรายงานการทดสอบ (Security Test Report) สรุปช่องโหว่ที่พบ วิธีการแก้ไข และองค์ความรู้ที่ได้รับจาก LAB เพื่อเตรียมความพร้อมสู่การทำงานจริงในสายงานวิศวกรรมซอฟต์แวร์

Step 2: สร้างและเตรียมฐานข้อมูล

สร้างฐานข้อมูลชื่อ `SecurityLab` พร้อมตารางและข้อมูลเริ่มต้นสำหรับทดสอบ โดยรันสคริปต์ SQL ที่ให้มาใน SQL Server Management Studio

database/setup.sql
CREATE DATABASE SecurityLab;
GO
USE SecurityLab;
GO

-- ตาราง Users
CREATE TABLE Users (
    id INT IDENTITY(1,1) PRIMARY KEY,
    username NVARCHAR(50) NOT NULL UNIQUE,
    password NVARCHAR(255) NOT NULL,
    email NVARCHAR(100) NOT NULL,
    role NVARCHAR(20) DEFAULT 'user'
);
GO

-- ตาราง Comments
CREATE TABLE Comments (
    id INT IDENTITY(1,1) PRIMARY KEY,
    user_id INT FOREIGN KEY REFERENCES Users(id),
    content NVARCHAR(MAX) NOT NULL,
    created_at DATETIME DEFAULT GETDATE()
);
GO

-- ตาราง Products
CREATE TABLE Products (
    id INT IDENTITY(1,1) PRIMARY KEY,
    name NVARCHAR(100) NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    description NVARCHAR(MAX)
);
GO

Step 3: สร้าง Backend (Vulnerable Version)

สร้างไฟล์ backend/server.js และใส่โค้ดฉบับเต็มต่อไปนี้ โค้ดนี้มีช่องโหว่ SQL Injection, XSS, และ IDOR

backend/server.js - VULNERABLE
// VULNERABLE VERSION
const express = require('express');
const sql = require('mssql');
const cors = require('cors');
const bodyParser = require('body-parser');

const app = express();
const PORT = 3000;

app.use(cors());
app.use(bodyParser.json());

const dbConfig = {
    user: 'your_username', // <-- แก้ไข
    password: 'your_password', // <-- แก้ไข
    server: 'localhost',
    database: 'SecurityLab',
    options: {
        encrypt: false,
        trustServerCertificate: true
    }
};

sql.connect(dbConfig).then(() => console.log('Connected to SQL Server')).catch(err => console.log('DB Connection Failed', err));

// 🚨 VULNERABLE: SQL Injection
app.post('/login', async (req, res) => {
    const { username, password } = req.body;
    const query = \`SELECT * FROM Users WHERE username='\${username}' AND password='\${password}'\`;
    const result = await sql.query(query);
    if (result.recordset.length > 0) {
        res.json({ success: true, user: result.recordset[0] });
    } else {
        res.status(401).json({ success: false, message: 'Invalid credentials' });
    }
});

// 🚨 VULNERABLE: XSS in comments
app.post('/comments', async (req, res) => {
    const { userId, content } = req.body;
    const query = \`INSERT INTO Comments (user_id, content) VALUES (\${userId}, '\${content}')\`;
    await sql.query(query);
    res.json({ success: true, message: 'Comment added' });
});

// 🚨 VULNERABLE: IDOR - Direct object reference
app.get('/user/:id', async (req, res) => {
    const userId = req.params.id;
    const query = \`SELECT * FROM Users WHERE id=\${userId}\`;
    const result = await sql.query(query);
    res.json(result.recordset[0] || {});
});

// Get all comments (for display)
app.get('/comments', async (req, res) => {
    const query = 'SELECT c.*, u.username FROM Comments c JOIN Users u ON c.user_id = u.id ORDER BY c.created_at DESC';
    const result = await sql.query(query);
    res.json(result.recordset);
});

app.listen(PORT, () => console.log(\`🚨 Vulnerable server running on http://localhost:\${PORT}\`));

const API_BASE = 'http://localhost:3000';
let currentUser = null;

// Login
document.getElementById('login-form').addEventListener('submit', async (e) => {
    e.preventDefault();

    const password = document.getElementById('password').value;

        method: 'POST',

    if (data.success) {
        currentUser = data.user;
        document.getElementById('login-result').innerHTML = \`Logged in as: <b>\${currentUser.username}</b>\`;

        document.getElementById('login-result').innerHTML = 'Login failed.';
    }
});

// Get User Profile
async function fetchUserProfile() {
    const userId = document.getElementById('user-id-input').value;
    const res = await fetch(\`\${API_BASE}/user/\${userId}\`);
    const data = await res.json();
    document.getElementById('profile-result').innerHTML = \`<pre>\${JSON.stringify(data, null, 2)}</pre>\`;
}

// Comments
document.getElementById('comment-form').addEventListener('submit', async (e) => {

สวมบทบาทแฮกเกอร์: ทดลองโจมตีช่องโหว่

หลังจากรัน Vulnerable App แล้ว ให้ลองใช้ Payloads ต่อไปนี้เพื่อเจาะระบบ

ทดสอบ SQL Injection

ที่หน้า Login, ลองกรอกข้อมูลต่อไปนี้:

  • Username: admin'; --
  • Password: (กรอกอะไรก็ได้)

ผลที่คาดว่าจะเกิด: คุณจะสามารถ Login เข้าระบบในฐานะ admin ได้สำเร็จ แม้จะไม่รู้รหัสผ่าน

ทดสอบ Cross-Site Scripting (XSS)

หลังจาก Login, ไปที่หน้า Comment แล้วโพสต์ข้อความต่อไปนี้:

<script>alert('XSS Attack! Your session could be stolen.');</script>

ผลที่คาดว่าจะเกิด: ทุกครั้งที่หน้าเว็บโหลด Comment นี้ขึ้นมา จะมีกล่องข้อความ (Alert Box) แสดงขึ้นมา ซึ่งพิสูจน์ว่าสามารถรัน JavaScript ในหน้าเว็บได้

ทดสอบ Insecure Direct Object Reference (IDOR)

ที่ช่อง User Profile, ลองใส่ User ID ของคนอื่น เช่น 1, 2, หรือ 3

ผลที่คาดว่าจะเกิด: คุณจะสามารถเห็นข้อมูลโปรไฟล์ทั้งหมดของผู้ใช้อื่นได้ (รวมถึงรหัสผ่านที่เก็บเป็น Plain text) โดยไม่มีการตรวจสอบสิทธิ์ใดๆ

🛡️ การป้องกัน SQL Injection: ใช้ Prepared Statements

เราจะใช้ Parameterized Queries ซึ่งจะแยกคำสั่ง SQL ออกจากข้อมูล (Data) อย่างชัดเจน ทำให้ Database Engine รู้ว่าส่วนไหนคือคำสั่ง และส่วนไหนคือข้อมูลที่ต้องเปรียบเทียบเท่านั้น วิธีนี้จะป้องกันการแทรกโค้ด SQL ได้อย่างสมบูรณ์

backend/secure-server.js - SECURE
// ✅ ใช้ Prepared Statement เพื่อความปลอดภัย
const request = new sql.Request();
request.input('username', sql.NVarChar, username);

// ข้อมูล @username จะถูกส่งไปอย่างปลอดภัย ไม่สามารถแทรกโค้ดได้
const result = await request.query('SELECT * FROM Users WHERE username = @username');

const user = result.recordset[0];
// จากนั้นค่อยเปรียบเทียบ password ที่ผ่านการ hash แล้วด้วย bcrypt
const isValidPassword = await bcrypt.compare(password, user.password);
if(isValidPassword) {
    // Login successful
}

🛡️ การป้องกัน XSS & IDOR

ใช้หลักการ "Defense in Depth" คือป้องกันหลายชั้น: ตรวจสอบและกรองข้อมูลที่รับเข้ามา (Input Validation), แปลงอักขระพิเศษให้ปลอดภัยก่อนแสดงผล (Output Encoding), และที่สำคัญที่สุดคือการตรวจสอบสิทธิ์ (Authorization) ทุกครั้งที่มีการร้องขอข้อมูล

backend/secure-server.js - SECURE
// Middleware ตรวจสอบ Token
function authenticateToken(req, res, next) {
    const token = req.headers['authorization']?.split(' ')[1];
    if (!token) return res.sendStatus(401);
    jwt.verify(token, JWT_SECRET, (err, user) => {
        if (err) return res.sendStatus(403);
        req.user = user;
        next();
    });
}

// ✅ ป้องกัน IDOR โดยตรวจสอบสิทธิ์ก่อนดึงข้อมูล
app.get('/user/:id', authenticateToken, async (req, res) => {
    const requestedUserId = parseInt(req.params.id);
    // User ทั่วไปดูได้แค่โปรไฟล์ตัวเอง แต่ Admin ดูได้ทุกคน
    if (req.user.userId !== requestedUserId && req.user.role !== 'admin') {
        return res.status(403).json({ error: 'Access denied' });
    }
    // ... ดึงข้อมูลด้วย Prepared statement ...
});

// ✅ ป้องกัน XSS โดยการ Encode ข้อมูลก่อนบันทึก
app.post('/comments', authenticateToken, async (req, res) => {
    const sanitizedContent = htmlEncode(req.body.content);
    // ... บันทึกข้อมูลที่ sanitized แล้วลง DB ...
});

เกณฑ์การประเมินผล

คะแนนจะถูกประเมินจากความเข้าใจในช่องโหว่, ความสามารถในการแก้ไขปัญหา, การวิเคราะห์เปรียบเทียบ, และการประยุกต์ใช้ความรู้

เกณฑ์การประเมิน คะแนน
1. การเข้าใจและทดสอบช่องโหว่ (Vulnerability Understanding) 30%
2. การแก้ไขช่องโหว่ (Vulnerability Mitigation) 40%
3. การวิเคราะห์และเปรียบเทียบ (Analysis & Comparison) 20%
4. การประยุกต์ใช้และตอบคำถาม (Application & Further Study) 10%

Deliverables: สิ่งที่ต้องส่ง

  • Source Code ทั้งเวอร์ชัน Vulnerable และ Secure
  • รายงานการทดสอบ (Security Test Report) ที่สรุปช่องโหว่, วิธีการโจมตี, และวิธีการแก้ไข
  • วิดีโอสาธิตการโจมตีและการป้องกัน (ถ้ามี)
